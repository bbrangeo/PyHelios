210class SphericalCoord(ctypes.Structure):
211 _fields_ = [
212 ('radius', ctypes.c_float),
213 ('elevation', ctypes.c_float),
214 ('zenith', ctypes.c_float),
215 ('azimuth', ctypes.c_float)
216 ]
217
218 def __repr__(self) -> str:
219 return f'SphericalCoord({self.radius}, {self.elevation}, {self.zenith}, {self.azimuth})'
220
221 def __str__(self) -> str:
222 return f'SphericalCoord({self.radius}, {self.elevation}, {self.zenith}, {self.azimuth})'
223
224 def __init__(self, radius:float=1, elevation:float=0, azimuth:float=0):
225 """
226 Initialize SphericalCoord matching C++ constructor.
227
228 Args:
229 radius: Radius (default: 1)
230 elevation: Elevation angle in radians (default: 0)
231 azimuth: Azimuthal angle in radians (default: 0)
232
233 Note: zenith is automatically computed as (/2 - elevation) to match C++ behavior
234 """
235 import math
236 self.radius = radius
237 self.elevation = elevation
238 self.zenith = 0.5 * math.pi - elevation # zenith = /2 - elevation (matches C++)
239 self.azimuth = azimuth
240
241 def from_list(self, input_list:List[float]):
242 self.radius = input_list[0]
243 self.elevation = input_list[1]
244 self.zenith = input_list[2]
245 self.azimuth = input_list[3]
246
247 def to_list(self) -> List[float]:
248 return [self.radius, self.elevation, self.zenith, self.azimuth]

46 def initialize(self, force_refresh: bool = False) -> None:
47 """
48 Initialize the plugin registry by scanning available plugins.
49
50 Args:
51 force_refresh: Force re-scanning even if already initialized
52 """
53 if self._initialized and not force_refresh:
54 return
55
56 try:
57 logger.info("Scanning for available plugins...")
58
59 # Detect available plugins
60 available_list = detect_available_plugins()
61 self._available_plugins = set(available_list)
62
63 # Get plugin capabilities
64 self._plugin_capabilities = get_plugin_capabilities()
65
66 # Clear failed plugins on successful initialization
67 self._failed_plugins.clear()
68
69 self._initialized = True
70
71 logger.info(f"Plugin registry initialized with {len(self._available_plugins)} plugins: {sorted(self._available_plugins)}")
72
73 except Exception as e:
74 logger.error(f"Failed to initialize plugin registry: {e}")
75 # Initialize with empty sets so the system can still function
76 self._available_plugins = set()
77 self._plugin_capabilities = {}
78 self._initialized = True

126 def require_plugin(self, plugin_name: str, feature_description: str = None) -> None:
127 """
128 Require that a plugin is available, raising an error if not.
129
130 Args:
131 plugin_name: Name of the required plugin
132 feature_description: Description of the feature requiring the plugin
133
134 Raises:
135 PluginNotAvailableError: If plugin is not available
136 """
137 if not self.is_plugin_available(plugin_name):
138 feature_text = f" for {feature_description}" if feature_description else ""
139
140 # Provide helpful error message with plugin information
141 error_msg = f"Plugin '{plugin_name}' is required{feature_text} but is not available."
142
143 if plugin_name in PLUGIN_METADATA:
144 metadata = PLUGIN_METADATA[plugin_name]
145 error_msg += f"\n\nPlugin Info:"
146 error_msg += f"\n- Description: {metadata.description}"
147 if metadata.gpu_required:
148 error_msg += f"\n- Requires GPU support"
149 if metadata.system_dependencies:
150 error_msg += f"\n- System dependencies: {', '.join(metadata.system_dependencies)}"
151
152 error_msg += f"\n\nTo enable this plugin:"
153 error_msg += f"\n1. Build PyHelios with the plugin: build_scripts/build_helios --plugins {plugin_name}"
154 error_msg += f"\n2. Or use a profile that includes it: build_scripts/build_helios --profile research"
155
156 # Mark as failed for future reference
157 self._failed_plugins.add(plugin_name)
158
159 raise PluginNotAvailableError(error_msg)

209 def get_plugin_summary(self) -> Dict[str, Any]:
210 """

223 'total_plugins': len(all_plugins),

226 'failed_plugins': len(self._failed_plugins),
227 'available_list': sorted(list(available)),
228 'missing_list': sorted([p for p in all_plugins if p not in available]),
229 'failed_list': sorted(list(self._failed_plugins)),
230 'gpu_plugins_available': len([p for p in available if PLUGIN_METADATA.get(p, {}).get('gpu_required', False)]),
231 'capabilities': self._plugin_capabilities or {}

282def graceful_plugin_fallback(plugin_name: str, fallback_result: Any = None,
283 warning_message: str = None):
284 """
285 Decorator to provide graceful fallback when a plugin is not available.
286
287 Args:
288 plugin_name: Name of the required plugin
289 fallback_result: Result to return if plugin is not available
290 warning_message: Custom warning message to display
291 """
292 def decorator(func: Callable) -> Callable:
293 @wraps(func)
294 def wrapper(*args, **kwargs):
295 registry = get_plugin_registry()
296
297 if not registry.is_plugin_available(plugin_name):
298 message = warning_message or f"Plugin '{plugin_name}' not available for {func.__name__}"
299 logger.warning(message)
300
301 # Suggest alternatives if available
302 alternatives = registry.suggest_alternatives(plugin_name)
303 if alternatives:
304 logger.info(f"Available alternatives: {alternatives}")
305
306 return fallback_result
307
308 return func(*args, **kwargs)
309 return wrapper
310 return decorator
